安裝基礎包

sudo pacman -S hyprland hyprpaper hyprlock rofi-wayland waybar mako wl-clipboard cliphist grim slurp xorg-xwayland xorg xdg-desktop-portal-hyprland xdg-desktop-portal-gtk

~/.config/hypr/
~/.config/waybar/
~/.config/mako/config

配置 ocr

安裝 tesseract

sudo pacman -S imagemagick tesseract-data-eng tesseract-data-chi_sim

配置 shell 腳本

~/scripts/ocr-hyprland.sh

#!/bin/env zsh 
SCR="$HOME/Documents/ocr_temp/src"
SCR2="$HOME/Documents/ocr_temp/src2"

# take a shot what you wana to OCR to text
grim -g "$(slurp)" $SCR.png

# increase the png
mogrify -modulate 100,0 -resize 400% $SCR.png
# should increase detection rate

# OCR by tesseract
tesseract $SCR.png $SCR &> /dev/null -l eng+chi_sim

# get the text and copy to clipboard

# sed -i 's/[[:space:]]//g' $SCR.txt # 删除空格方式1
# sed -i 's/\ //g' $SCR.txt  # 删除空格方式2

cat $SCR.txt  | sed -r 's/([^0-9a-z])?\s+([^0-9a-z])/\1\2/ig'>$SCR2.txt  # 解决每个汉字之间有空格的情况，英文单词间空格依旧保留
cat $SCR2.txt | wl-copy

kwrite $SCR2.txt
exit

~/.config/hypr/hyprland.conf

bind = $mainMod SHIFT, O, exec, bash "~/scripts/ocr-hyprland.sh"

env = XDG_MENU_PREFIX,plasma-
