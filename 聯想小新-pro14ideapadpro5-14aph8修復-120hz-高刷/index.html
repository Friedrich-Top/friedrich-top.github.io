  1. 转换为二进制文件
1

  1. 添加至驱动固件
1

sudo mkdir /usr/lib/firmware/edid

1

sudo cp edid-fixed.bin /usr/lib/firmware/edid
  1. 配置初始 ramdisk 环境 /etc/mkinitcpio.conf
1
2
3

...

...
  1. 生成新 initramfs 映像
1

sudo mkinitcpio -P
  1. 配置默认内核参数 /etc/default/grub
1

GRUB_CMDLINE_LINUX_DEFAULT="loglevel=5 nowatchdog drm.edid_firmware=eDP-1:edid/edid-fixed.bin"
  1. 更新 grub
1

sudo grub-mkconfig -o /boot/grub/grub.cfg
  1. 重启系统
1

reboot
